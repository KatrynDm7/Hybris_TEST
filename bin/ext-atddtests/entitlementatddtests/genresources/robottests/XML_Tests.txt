Test_Xml_Equal
    [Documentation]    test if two xml snippets are equal
    ${actualXml}=    Set Variable    <outer id="1234"><inner><id>5678</id></inner></outer>
    ${expectedXml}=    Set Variable    <outer id="1234"><inner><id>5678</id></inner></outer>    
    verify XML "${actualXml}" equals expected XML "${expectedXml}"
